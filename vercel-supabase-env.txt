# 🚀 Vercel + Supabase 环境变量配置模板
# 将这些变量添加到 Vercel 项目的 Environment Variables 中

# ===========================================
# 基础应用配置
# ===========================================
APP_NAME=独角数卡
APP_ENV=production
APP_DEBUG=false
APP_URL=https://your-project.vercel.app

# Laravel 应用密钥（运行 php artisan key:generate 生成）
APP_KEY=base64:your-generated-key-here

# ===========================================
# Supabase PostgreSQL 数据库配置
# ===========================================
DB_CONNECTION=pgsql
DB_HOST=db.xxxxxx.supabase.co
DB_PORT=5432
DB_DATABASE=postgres
DB_USERNAME=postgres
DB_PASSWORD=your-supabase-password

# Supabase 项目配置
SUPABASE_URL=https://xxxxxx.supabase.co
SUPABASE_KEY=your-anon-key
SUPABASE_SECRET=your-service-role-key
SUPABASE_DB_HOST=db.xxxxxx.supabase.co
SUPABASE_DB_PORT=5432
SUPABASE_DB_DATABASE=postgres
SUPABASE_DB_USERNAME=postgres
SUPABASE_DB_PASSWORD=your-supabase-password

# ===========================================
# Vercel 优化配置
# ===========================================
VERCEL=1
CACHE_DRIVER=array
SESSION_DRIVER=cookie
SESSION_LIFETIME=120
QUEUE_CONNECTION=sync
LOG_CHANNEL=stderr
LOG_LEVEL=error
FILESYSTEM_DISK=local

# ===========================================
# 路径配置（Vercel 专用）
# ===========================================
VIEW_COMPILED_PATH=/tmp/views
STORAGE_PATH=/tmp/storage
CACHE_PATH=/tmp/cache
SESSION_PATH=/tmp/sessions

# ===========================================
# 禁用不必要的服务
# ===========================================
BROADCAST_DRIVER=null

# ===========================================
# 邮件配置（可选）
# ===========================================
MAIL_MAILER=smtp
MAIL_HOST=smtp.gmail.com
MAIL_PORT=587
MAIL_USERNAME=your-email@gmail.com
MAIL_PASSWORD=your-app-password
MAIL_ENCRYPTION=tls
MAIL_FROM_ADDRESS=your-email@gmail.com
MAIL_FROM_NAME=独角数卡

# ===========================================
# 独角数卡特定配置
# ===========================================
DUJIAO_ADMIN_LANGUAGE=zh_CN

# ===========================================
# 性能优化配置
# ===========================================
APP_CONFIG_CACHE=/tmp/config.php
APP_ROUTES_CACHE=/tmp/routes.php
APP_PACKAGES_CACHE=/tmp/packages.php
APP_SERVICES_CACHE=/tmp/services.php
